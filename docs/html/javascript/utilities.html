Format Strings

template = 'https://www.reddit.com/submit?url=${url}&title=${title}';
data = {
  url: 'http://arxitics.com',
  title: 'arXiv Analytics'
};
schema.format(template, data) => 'https://www.reddit.com/submit?url=http://arxitics.com&title=arXiv Analytics'
